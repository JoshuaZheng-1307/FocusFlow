    <script>
        // 1. 头像预览功能
        function previewImage(input) {
            if (input.files && input.files[0]) {
                var reader = new FileReader();
                reader.onload = function(e) {
                    document.getElementById('avatar-preview').src = e.target.result;
                    document.getElementById('avatar-preview').style.display = 'block';
                    document.getElementById('avatar-text').style.display = 'none';
                }
                reader.readAsDataURL(input.files[0]);
            }
        }

        // 2. 页面加载完成后请求数据
        document.addEventListener('DOMContentLoaded', function() {
            fetch('/api/home/data/')
                .then(response => response.json())
                .then(data => {
                    console.log("拿到数据了:", data);

                    // 2.1 更新积分
                    if(data.total_score !== undefined) {
                        document.getElementById('js-score').innerText = data.total_score;
                    }

                    // 2.2 更新好友列表
                    const friendListEl = document.getElementById('js-friend-list');
                    const noMsgEl = document.getElementById('no-friend-msg');

                    if(data.friends && data.friends.length > 0) {
                        let html = '';
                        data.friends.forEach(f => {
                            html += `
                                <div class="friend-item">
                                    <div class="friend-avatar"></div>
                                    <div class="friend-details">
                                        <h4>${f.name}</h4>
                                        <p>心流积分: ${f.score}</p>
                                    </div>
                                </div>`;
                        });
                        friendListEl.innerHTML = html;
                        document.getElementById('js-friend-count').innerText = data.friends.length;
                        noMsgEl.style.display = 'none'; // 隐藏无数据提示
                    } else {
                        noMsgEl.style.display = 'block'; // 显示无数据提示
                    }

                    // 2.3 更新排行榜
                    const rankListEl = document.getElementById('js-leaderboard-list');
                    if(data.leaderboard && data.leaderboard.length > 0) {
                        let html = '';
                        data.leaderboard.forEach((item, index) => {
                            html += `
                                <div class="rank-item">
                                    <div class="rank-num">${index + 1}</div>
                                    <div class="rank-avatar"></div>
                                    <div class="rank-info">
                                        <div class="rank-name">${item.user}</div>
                                        <div class="rank-score">心流积分: ${item.score}</div>
                                    </div>
                                </div>`;
                        });
                        rankListEl.innerHTML = html;
                    }
                })
                .catch(error => {
                    console.error('请求失败:', error);
                    // 即使 API 失败，也显示原本的 Django 数据（如果有的话）
                });
        });
    </script>
