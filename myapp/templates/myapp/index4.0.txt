   <script>
    // ==========================================
    // 1. 工具函数区 (Utility Functions)
    // ==========================================

    /**
     * 将秒数转换为 HH:MM:SS 格式
     */
    function formatTime(seconds) {
        const h = String(Math.floor(seconds / 3600)).padStart(2, '0');
        const m = String(Math.floor((seconds % 3600) / 60)).padStart(2, '0');
        const s = String(seconds % 60).padStart(2, '0');
        return `${h}:${m}:${s}`;
    }

    /**
     * 处理头像上传
     */
    function uploadAvatar(input) {
        if (input.files && input.files[0]) {
            const formData = new FormData();
            formData.append('avatar', input.files[0]);
            // 注意：确保你的 HTML 模板中有 {% csrf_token %} 或者这里能获取到 token
            const csrfToken = document.querySelector('[name=csrfmiddlewaretoken]')?.value || '{{ csrf_token }}';
            formData.append('csrfmiddlewaretoken', csrfToken);

            fetch('/upload-avatar/', {
                method: 'POST',
                body: formData,
                headers: {
                    'X-CSRFToken': csrfToken // 双重保险
                }
            })
            .then(response => response.json())
            .then(data => {
                if (data.status === 'success') {
                    const img = document.getElementById('avatar-preview');
                    const text = document.getElementById('avatar-text');
                    if(img && data.avatar_url) {
                        img.src = data.avatar_url + '?t=' + new Date().getTime(); // 加时间戳防止缓存
                        img.style.display = 'block';
                        if(text) text.style.display = 'none';
                    }
                } else {
                    alert('上传失败: ' + (data.message || '未知错误'));
                }
            })
            .catch(err => console.error('Upload error:', err));
        }
    }

    // ==========================================
    // 2. 核心业务逻辑 (Core Logic)
    // ==========================================

    document.addEventListener('DOMContentLoaded', function() {
        // 获取 DOM 元素引用
        const scoreEl = document.getElementById('js-score');
        const avatarImg = document.getElementById('avatar-preview');
        const avatarText = document.getElementById('avatar-text');

        // 排行榜容器 (请确保 HTML 中有 id="js-leaderboard-list")
        const leaderboardListEl = document.getElementById('js-leaderboard-list');
        // 成就/方向列表容器 (复用原来的 friend-list 或新建一个)
        const directionListEl = document.getElementById('js-friend-list');
        const directionCountEl = document.getElementById('js-friend-count');

        // 发起请求获取首页数据
        fetch('/api/home/data/')
            .then(response => {
                if (!response.ok) throw new Error('Network response was not ok');
                return response.json();
            })
            .then(data => {
                console.log("首页数据加载成功:", data);

                // --- A. 更新基础信息 ---
                if (scoreEl) scoreEl.innerText = data.total_score || 0;

                if (data.user && data.user.avatar) {
                    if (avatarImg) {
                        avatarImg.src = data.user.avatar;
                        avatarImg.style.display = 'block';
                    }
                    if (avatarText) avatarText.style.display = 'none';
                }

                // --- B. 渲染“我的学习方向” (Top 3 成就) ---
                // 假设后端返回了 directions 数组，包含 name, time_seconds, score
                // 如果后端没排好序，这里手动排序：按积分或时长降序
                let myDirections = data.directions || [];
                myDirections.sort((a, b) => b.score - a.score); // 按积分从高到低

                // 只取前 3 个用于展示（如果需要展示全部，去掉 slice）
                const top3Directions = myDirections.slice(0, 3);

                let dirHtml = '';
                if (top3Directions.length > 0) {
                    top3Directions.forEach((dir, index) => {
                        // 根据排名添加不同的样式类 (可选)
                        let rankClass = index === 0 ? 'rank-1' : (index === 1 ? 'rank-2' : 'rank-3');

                        dirHtml += `
                            <div class="friend-item ${rankClass}">
                                <div class="friend-rank">#${index + 1}</div>
                                <div class="friend-details">
                                    <h4>${dir.name}</h4>
                                    <p>心流积分: <span class="highlight">${dir.score}</span></p>
                                    <small>专注时长: ${formatTime(dir.time_seconds)}</small>
                                </div>
                            </div>`;
                    });
                } else {
                    dirHtml = '<div class="empty-tip">暂无学习记录，快去 FocusFlow 开始吧！</div>';
                }

                if (directionListEl) directionListEl.innerHTML = dirHtml;
                if (directionCountEl) directionCountEl.innerText = myDirections.length;


                // --- C. 渲染“用户排行榜” ---
                // 假设后端返回了 leaderboard 数组，包含 username, avatar, total_score
                let users = data.leaderboard || [];
                // 确保按总分降序
                users.sort((a, b) => b.total_score - a.total_score);

                let userHtml = '';
                users.forEach((u, index) => {
                    // 如果是当前登录用户，可以高亮显示
                    const isMe = u.is_me ? 'current-user' : '';

                    userHtml += `
                        <div class="leaderboard-item ${isMe}">
                            <div class="lb-rank">${index + 1}</div>
                            <img src="${u.avatar || '/static/img/default_avatar.png'}" class="lb-avatar" alt="avatar">
                            <div class="lb-info">
                                <span class="lb-name">${u.username}</span>
                            </div>
                            <div class="lb-score">${u.total_score} 分</div>
                        </div>
                    `;
                });

                // 请将 HTML 中对应的容器 ID 改为 js-leaderboard-list
                // 如果你还没有这个容器，请在 HTML 右侧区域添加一个 <ul id="js-leaderboard-list"></ul>
                const lbContainer = document.getElementById('js-leaderboard-list');
                if (lbContainer) {
                    lbContainer.innerHTML = userHtml;
                }
            })
            .catch(error => {
                console.error('获取首页数据失败:', error);
            });
    });
