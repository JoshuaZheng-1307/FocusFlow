        <div class="controls">
            <button class="action-btn btn-start" id="btnStart" onclick="toggleTimer()">开始学习</button>
            <button class="action-btn btn-stop" id="btnStop" onclick="stopAndSwitch()">√ 结束并切换</button>
            <button class="action-btn btn-master" id="btnMaster" onclick="masterDirection()">完全精通</button>
        </div>
        <div class="points-card">
            <div class="points-value" id="pointsDisplay">0</div>
            <div class="points-label">累计心流积分</div>

    <script>
        // --- State Management ---
        let directions = JSON.parse(localStorage.getItem('ff_directions')) || [
            { id: 1, name: 'Python 后端开发', time: 0, mastered: false },
            { id: 2, name: 'UI/UX 交互设计', time: 0, mastered: false }
        ];
        let activeId = null;
        let isRunning = false;
        let interval = null;
        let nextId = directions.length > 0 ? Math.max(...directions.map(d => d.id)) + 1 : 1;

        // --- Core Functions ---
        function saveState() {
            localStorage.setItem('ff_directions', JSON.stringify(directions));
        }

        function renderNav() {
            const navList = document.getElementById('navList');
            navList.innerHTML = '';
            directions.forEach(dir => {
                const li = document.createElement('li');
                li.className = `nav-item ${dir.id === activeId ? 'active' : ''} ${dir.mastered ? 'mastered' : ''}`;
                li.innerHTML = `<span>${dir.name}</span><span style="font-size:12px">${formatTime(dir.time)}</span>`;
                li.onclick = () => selectDirection(dir.id);
                navList.appendChild(li);
            });
            updatePoints();
        }

        function addDirection() {
            const name = prompt("请输入新的学习方向名称：");
            if (name && name.trim()) {
                directions.push({ id: nextId++, name: name.trim(), time: 0, mastered: false });
                saveState();
                renderNav();
            }
        }

        function selectDirection(id) {
            if (isRunning) return;
            const dir = directions.find(d => d.id === id);
            if (!dir || dir.mastered) return;
            activeId = id;
            document.getElementById('currentLabel').innerText = `当前专注：${dir.name}`;
            document.getElementById('timerDisplay').textContent = formatTime(dir.time);
            document.getElementById('btnStart').style.display = 'inline-block';
            document.getElementById('btnStop').style.display = 'none';
            document.getElementById('btnMaster').style.display = 'none';
            renderNav();
        }

        function toggleTimer() {
            if (!activeId) return alert("请先选择一个学习方向！");
            isRunning = true;
            document.getElementById('btnStart').style.display = 'none';
            document.getElementById('btnStop').style.display = 'inline-block';
            document.getElementById('btnMaster').style.display = 'inline-block';
            document.getElementById('timerDisplay').classList.add('running');
            interval = setInterval(() => {
                const dir = directions.find(d => d.id === activeId);
                if (dir) {
                    dir.time++;
                    document.getElementById('timerDisplay').textContent = formatTime(dir.time);
                    saveState();
                    const items = document.querySelectorAll('.nav-item');
                    items.forEach(item => {
                        if(item.classList.contains('active')) {
                            item.querySelector('span:last-child').textContent = formatTime(dir.time);
                        }
                    });
                    updatePoints();
                }
            }, 1000);
        }

        function stopAndSwitch() {
            clearInterval(interval);
            isRunning = false;
            document.getElementById('timerDisplay').classList.remove('running');
            document.getElementById('btnStart').style.display = 'inline-block';
            document.getElementById('btnStop').style.display = 'none';
            document.getElementById('btnMaster').style.display = 'none';
            document.getElementById('currentLabel').innerText = "选择下一个方向继续...";
            renderNav();
        }

        function masterDirection() {
            if (!activeId) return;
            if(confirm("确认该方向已完全精通？\n此操作将永久锁定该方向的时间，并获得额外1倍积分奖励！")) {
                const dir = directions.find(d => d.id === activeId);
                dir.mastered = true;
                saveState();
                stopAndSwitch();
                alert(`🎉 恭喜！"${dir.name}" 已达成精通成就！`);
            }
        }

        // --- Calculation Logic ---
        function updatePoints() {
            let totalUnlockedTimeSec = directions.filter(d => !d.mastered).reduce((sum, d) => sum + d.time, 0);
            let hours = totalUnlockedTimeSec / 3600;
            let basePoints = 0;
            if (hours > 0) {
                basePoints = Math.round(5 * Math.pow(hours, 1.5) + 20 * hours);
            }
            let masteryBonus = directions.filter(d => d.mastered).reduce((sum, d) => sum + d.time, 0);
            const finalPoints = basePoints + masteryBonus;
            document.getElementById('pointsDisplay').textContent = finalPoints.toLocaleString();
        }

        // --- Utility Functions ---
        function formatTime(seconds) {
            const h = String(Math.floor(seconds / 3600)).padStart(2, '0');
            const m = String(Math.floor((seconds % 3600) / 60)).padStart(2, '0');
            const s = String(seconds % 60).padStart(2, '0');
            return `${h}:${m}:${s}`;
        }

        // --- New Functions ---
        function toggleNightMode() {
            document.body.classList.toggle('night-mode');
        }

        function logout() {
            if (confirm("确定要退出学习吗？")) {
                // 这里可以添加跳转逻辑，目前仅重置状态
                window.location.reload();
            }
        }

        // Init
        renderNav();
    </script>
