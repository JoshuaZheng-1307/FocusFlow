def register(request):
    """注册视图"""
    if request.method == 'POST':
        form = RegisterForm(request.POST)
        if form.is_valid():
            user = form.save()
            login(request, user)
            # 【关键点】注册成功后跳转到 index (即 dashboard)
            return redirect('myapp:index')
    else:
        form = RegisterForm()
    return render(request, 'myapp/register.html', {'form': form})
